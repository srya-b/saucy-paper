\begin{lstlisting}[basicstyle=\small\ttfamily,frame=single]
stype sender = +{ commit : bit ^ +{ open : 1 } } 

stype receiver = &{ commit : &{ open : bit -> 1 } }

decl F_comm : (S : sender), (R : receiver) |- (F : 1)

def F <- F_comm S R =
  case s (
    commit => b = recv S ;
              R.commit ;
              case S (
                open => R.open ;
                        send R b ;
                        wait S ; wait R ; close F ) )
\end{lstlisting}
